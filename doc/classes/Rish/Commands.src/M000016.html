    def help(word=nil)
      if word
        puts %x{qri #{word}}
      else
        puts "Rish builtin-commands\n\n@p <expr>  profile expression (requires --debug on JRuby)\n?<expr>    show help for command expr (doesn't work so well on IRB :( )\n@h         show this help\n!<cmd>     Run cmd in a shell\n!<cmd> &   Run cmd in background \n!&<cmd>    Same as above (version for IRB which doesn't like an \"&\" \n           as last element on line)\n"
      end
    end
